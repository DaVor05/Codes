public class MezclaDirecta {

    // Método principal para la ordenación por mezcla
    public static void mergeSort(int[] arr, int left, int right) {
        if (left < right) {
            int middle = (left + right) / 2;

            System.out.print("Dividiendo: ");
            printArray(arr, left, right);

            // Ordenar la primera y segunda mitad
            mergeSort(arr, left, middle);
            mergeSort(arr, middle + 1, right);

            // Combinar las dos mitades ordenadas
            merge(arr, left, middle, right);

            System.out.print("Combinando: ");
            printArray(arr, left, right);
        }
    }

    // Método para combinar dos subarreglos
    public static void merge(int[] arr, int left, int middle, int right) {
        int n1 = middle - left + 1;
        int n2 = right - middle;

        // Crear arreglos temporales
        int[] leftArray = new int[n1];
        int[] rightArray = new int[n2];

        // Copiar datos a los arreglos temporales
        for (int i = 0; i < n1; i++) {
            leftArray[i] = arr[left + i];
        }
        for (int j = 0; j < n2; j++) {
            rightArray[j] = arr[middle + 1 + j];
        }

        // Combinar los arreglos temporales
        int i = 0, j = 0;
        int k = left;
        while (i < n1 && j < n2) {
            if (leftArray[i] <= rightArray[j]) {
                arr[k] = leftArray[i];
                i++;
            } else {
                arr[k] = rightArray[j];
                j++;
            }
            k++;
        }

        // Copiar los elementos restantes de leftArray, si hay alguno
        while (i < n1) {
            arr[k] = leftArray[i];
            i++;
            k++;
        }

        // Copiar los elementos restantes de rightArray, si hay alguno
        while (j < n2) {
            arr[k] = rightArray[j];
            j++;
            k++;
        }
    }

    // Método para imprimir el arreglo en un rango específico
    public static void printArray(int[] arr, int left, int right) {
        for (int i = left; i <= right; i++) {
            System.out.print(arr[i] + " ");
        }
        System.out.println();
    }

    public static void main(String[] args) {
        int[] ventasDiarias = {300, 1500, 200, 500, 1200, 700, 50};

        System.out.println("Arreglo original:");
        printArray(ventasDiarias, 0, ventasDiarias.length - 1);

        // Llamar al método de ordenación
        mergeSort(ventasDiarias, 0, ventasDiarias.length - 1);

        System.out.println("Arreglo ordenado:");
        printArray(ventasDiarias, 0, ventasDiarias.length - 1);
    }
}
